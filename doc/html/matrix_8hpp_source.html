221 bool Matrix::equals(Matrix other) {
222  // no need for further computation if dimensions are not identical
223  if (rows != other.rows || cols != other.cols) {
224  return false;
225  }
226  // assume the matrices are equal until proven otherwise
227  bool truth = true;
228 
229  // extra loop condition to account for inner break statement
230  for (int i = 0; (i < rows) && truth; i++) {
231  for (int j = 0; j < cols; j++) {
232  // matrix inequality proven, no need to continue
233  if (matrix[i][j] != other.matrix[i][j]) {
234  truth = false;
235  break;
236  }
237  }
238  }
239  return truth;

302  std::thread threads[nThreads];
303

335 void Matrix::multLogic(Matrix *a, Matrix *bT, Matrix *product, int offset, int skip) {
336  for (int i = offset; i < a->getRows(); i+=skip) {
337  // point to ith rows of a and product matrices
338  double *ithRowA = a->linkRow(i);
339  double *ithRowProduct = product->linkRow(i);
340 
341  for (int j = 0; j < bT->getRows(); j++) {
342  // point to the jth row of bT
343  double *jthRowBT = bT->linkRow(j);
344  // initialize sum to zero
345  ithRowProduct[j] = 0;
346 
347  for (int k = 0; k < a->getCols(); k++) {
348  // equivalent to C(i, j) += A(i, k) * B^T(j, k)
349  // or also to C(i, j) += A(i, k) * B(k, j)
350  ithRowProduct[j] += ithRowA[k] * jthRowBT[k];
351  }
352  }
353  }
354 }
